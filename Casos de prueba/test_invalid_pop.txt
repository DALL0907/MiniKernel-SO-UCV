// 1. LOAD 10 (Operación sencilla previa)
04100010  // LOAD inmediato 10 -> AC = 10

// 2. STR 50 (Guardamos el 10 en memoria[50])
05000050  // STR directo en dirección 50

// 3. LOAD 999 (Otro dato cualquiera)
04100999  // LOAD inmediato 999 -> AC = 999

// 4. PSH 100 (Hacemos un PUSH para tener algo en la pila)
25100100  // PSH inmediato 100 -> Stack: [100]

// 5. POP 60 (Sacamos el 100 y lo guardamos en memoria[60])
26000060  // POP directo a dirección 60 -> Stack vacía

// 6. POP 70 (¡ERROR! Underflow - intentamos POP con pila vacía)
26000070  // POP directo a dirección 70 -> Genera INT_UNDERFLOW (código 7)
